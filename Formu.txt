import javax.swing.*; //importacion de libreria swingpublic class Formu extends JFrame{    private JLabel label1; //creacion de etiqueta 1  private JLabel label2; //creacion de etiqueta 2    public Formu(){ //Constructor para personalizar etiquetas    setLayout(null);    label1 = new JLabel("Interfaz grafica."); //mostrar texto en etiqueta    label1.setBounds(10, 20, 300, 30); //posicion de etiqueta    add(label1); //mostrar todo     label2 = new JLabel("Version 1.0"); //mostrar texto en etiqueta    label2.setBounds(10, 100, 100, 30 ); //posicion de etiqueta    add(label2); //mostrar todo   }    public static void main(String args[]){    Formu formulario1 = new Formu(); //creacion de interfaz    formulario1.setBounds(0, 0, 300, 200); //dimension de interfaz    formulario1.setResizable(false); //no deja que el usuario manipule las medidas del Frame    formulario1.setVisible(true); //ver el Frame    formulario1.setLocationRelativeTo(null); //posicionar en el centro de la pantalla  }}
