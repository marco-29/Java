import java.util.Scanner;public class VectorDi{  public static void main(String args[]){    Scanner morita = new Scanner(System.in);   int longitud = 0;   System.out.print("Ingrese los numeros deseados: ");   longitud = morita.nextInt();   int numeros[] = new int[longitud];      for(int i = 0; i < numeros.length; i++){ //length sirve para saver la longitud del vector     System.out.println("Ingrese el valor #" + (i + 1));     numeros[i] = morita.nextInt();   }   for(int i = 0; i < numeros.length; i++){     System.out.print("[" + numeros[i] + "] ");   } }}
